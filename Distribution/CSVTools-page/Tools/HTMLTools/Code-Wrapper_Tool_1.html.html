const inputCode = document.getElementById('inputCode');
const outputCode = document.getElementById('outputCode');
const maxWidthInput = document.getElementById('maxWidth');
const copyBtn = document.getElementById('copyBtn');
const copyText = document.getElementById('copyText');
const widthDisplay = document.getElementById('widthDisplay');
const widthInfo = document.getElementById('widthInfo');
maxWidthInput.addEventListener('input', function() {
const width = this.value || 90;
widthDisplay.textContent = width;
widthInfo.textContent = width;
});
outputCode.addEventListener('input', function() {
copyBtn.style.display = this.value ? 'flex' : 'none';
});
function wrapLines() {
const code = inputCode.value;
const maxWidth = parseInt(maxWidthInput.value) || 90;
if (!code) {
alert('Please paste some code first!');
return;
}
const lines = code.split('\n');
const result = [];
lines.forEach(line => {
const indentMatch = line.match(/^(\s*)/);
const indent = indentMatch ? indentMatch[1] : '';
const indentLength = indent.length;
if (line.length <= maxWidth) {
result.push(line);
return;
}
let remaining = line;
let isFirstLine = true;
while (remaining.length > maxWidth) {
const effectiveWidth = maxWidth;
let breakPoint = effectiveWidth;
const searchStart = Math.max(0, effectiveWidth - 30);
const breakChars = [' ', ',', ';', '.', '(', ')', '{', '}', '[', ']',
'>', '<', '+', '-', '='];
let bestBreak = -1;
for (let i = effectiveWidth; i >= searchStart; i--) {
if (breakChars.includes(remaining[i])) {
bestBreak = i + 1;
break;
}
}
if (bestBreak > indentLength) {
breakPoint = bestBreak;
}
const currentLine = remaining.substring(0, breakPoint).trimEnd();
result.push(currentLine);
remaining = indent + remaining.substring(breakPoint).trimStart();
isFirstLine = false;
}
if (remaining.trim()) {
result.push(remaining);
}
});
outputCode.value = result.join('\n');
}
function copyOutput() {
outputCode.select();
document.execCommand('copy');
copyText.textContent = 'Copied!';
copyBtn.classList.add('copied');
setTimeout(() => {
copyText.textContent = 'Copy';
copyBtn.classList.remove('copied');
}, 2000);
}
function clearAll() {
inputCode.value = '';
outputCode.value = '';
}

}

}
